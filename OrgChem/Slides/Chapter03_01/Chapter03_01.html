一种判定基团优先级的递归伪码

function CompareGroupPriority(GroupA, GroupB: TGroup): Integer;

begin

  Result := 0; 

  if FirstAtom(GroupA) <> FirstAtom(GroupB) then 

    Result := FirstAtom(GroupA) - FirstAtom(GroupB)

  else begin 

    // Compare first group on first atom of GroupA/B 

    Result := CompareGroupPriority(FirstGroup(GroupA), 
　　FirstGroup(GroupB));

    if Result = 0 then   // Same first group 

    begin 

      // Compare second group on first atom of GroupA/B 

      Result := CompareGroupPriority(SecondGroup(GroupA), 
　　　SecondGroup(GroupB));

      if Result = 0 then  // Same second group 

        Result := CompareGroupPriority(ThirdGroup(GroupA), 
　　　　ThirdGroup(GroupB));

    end;

  end;

end;
